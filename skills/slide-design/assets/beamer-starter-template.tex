% Frame title: Mondrian grid-cell header, kept quiet (chrome is decoration --
% pleasing but quiet). Small blue cell bounded by the black grid (2.2pt vertical
% line + full-width 1.6pt rule); flat red plane (title-page proportions,
% 4.85em x 0.87ex) sitting ON the rule at the right margin. The red is anchored
% to the RULE (zero-height box drawn just above it), not to the title line, so
% it stays put when a long title wraps; \hangindent aligns continuation lines
% with the title text. Box depth 0.9ex keeps smallcaps-title parentheses clear
% of the rule while staying tight.
% Wrapped in a macro because metropolis's \appendix hook (progressbar=none)
% resets headline/frametitle/footline to plain -- re-assert after \appendix
% (see the appendix section in the document body).
\newcommand{\MondrianFrametitle}{%
\setbeamertemplate{frametitle}{%
  \nointerlineskip
  \vspace{0.6em}
  \begin{beamercolorbox}[wd=\textwidth, sep=0pt, dp=0.9ex]{frametitle}%
    \hangindent=\dimexpr0.9em+2.2pt+1.0ex\relax
    {\color{MondrianBlue}\rule[-0.9ex]{0.9em}{3.4ex}}%
    {\color{MondrianBlack}\rule[-0.9ex]{2.2pt}{3.4ex}}\hspace{1.0ex}%
    {\usebeamerfont{frametitle}\insertframetitle\strut}%
  \end{beamercolorbox}
  \nointerlineskip
  \hbox to \textwidth{\hfill\smash{\color{MondrianRed}\rule[0pt]{4.85em}{0.87ex}}}%
  \nointerlineskip
  {\color{MondrianBlack}\rule{\textwidth}{1.6pt}}
  \vspace{-1em}
}}
\MondrianFrametitle
\setbeamertemplate{background}{}

  \begin{beamercolorbox}[wd=\textwidth, sep=0pt]{title}

    {\color{MondrianBlue}\rule{3em}{1.4ex}}\hfill{\color{MondrianRed}\rule{4.85em}{0.87ex}}\par\nointerlineskip% phi-wider, phi-flatter red: equal area, asymmetric shape

% Itemize: neutral grayscale squares (darker -> lighter for nesting).
% Markers stay neutral so red/blue in the text always carry meaning;
% the chrome (blue + red title blocks) keeps color on every slide.
% baseline=-0.4ex vertically centers the square on the text line.
\setbeamertemplate{itemize item}{%
\tikz[baseline=-0.4ex]{\fill[MondrianBlack] (0,0) rectangle (0.65ex,0.65ex);}}
\setbeamertemplate{itemize subitem}{%
\tikz[baseline=-0.4ex]{\fill[MondrianBlack!55] (0,0) rectangle (0.55ex,0.55ex);}}
\setbeamertemplate{itemize subsubitem}{%
\tikz[baseline=-0.4ex]{\fill[MondrianBlack!35] (0,0) rectangle (0.45ex,0.45ex);}}

% Enumerate: numbered diamonds (neutral; (n.base) aligns the number with the text baseline)
\setbeamertemplate{enumerate item}{%
\tikz[baseline=(n.base)]\node[shape=diamond, inner sep=0.15ex, minimum size=1.35ex, fill=MondrianBlack, text=white, font=\bfseries\scriptsize, align=center](n){\insertenumlabel};}
\setbeamertemplate{enumerate subitem}{%
\tikz[baseline=(n.base)]\node[shape=diamond, inner sep=0.12ex, minimum size=1.15ex, fill=MondrianBlack!65, text=white, font=\scriptsize, align=center](n){\insertsubenumlabel};}
\setbeamertemplate{enumerate subsubitem}{%
\tikz[baseline=(n.base)]\node[shape=diamond, inner sep=0.10ex, minimum size=1.00ex, draw=MondrianBlack, text=MondrianBlack, font=\tiny, align=center](n){\insertsubsubenumlabel};}

%% ============================================================
%% OVERLAY COMMAND REFERENCE
%% Beamer overlay specifications control when content appears
%% ============================================================
%%
%% BASIC OVERLAY SPECS:
%%   <N>       - Show only on slide N
%%   <N->      - Show from slide N onwards
%%   <-N>      - Show up to slide N
%%   <N-M>     - Show from slide N to M
%%
%% INCREMENTAL REVEAL (apply per list: \begin{itemize}[<+->]):
%%   <+->      - Appear on "next" slide and stay visible
%%   <+(1)->   - Skip one step, then appear (offset by 1)
%%   <.->      - Appear on SAME slide as previous <+-> item (no increment)
%%
%% CONTENT VISIBILITY:
%%   \pause        - Simple pause; advances overlay counter
%%   \onslide<N>   - Content visible on slide N (takes space always)
%%   \only<N>      - Content exists only on N (no space otherwise)
%%   \visible<N>   - Like \onslide but can span blocks
%%   \uncover<N>   - Shows content on N, hidden (but spaced) otherwise
%%
%% HIGHLIGHTING:
%%   \alert<N>{text}   - Highlight text on slide N (uses alerted text color)
%%   \alert<N,M>{text} - Highlight on multiple specific slides
%%
%% ADVANCED:
%%   <+-| alert@+>     - Reveal incrementally AND alert on appearance
%%   <only@N>          - Alternative syntax for \only<N> on \item
%%   <visible@N>       - Alternative syntax for \visible<N> on \item
%%
%% FRAME LABELS AND REUSE:
%%   [label=name]      - Give frame a label for later reference
%%   \againframe<N>{name} - Show slide N of labeled frame again
%%   \slidelink{label}{text} - Clickable link to labeled frame
%%
%% OVERPRINT ENVIRONMENT:
%%   \begin{overprint} - Stack content in same space across overlays
%%     \onslide<N> content
%%   \end{overprint}
%%
%% ============================================================

% <1-3> limits this frame to overlays 1-3; [label=roadmap] enables \againframe returns
\begin{frame}<1-3>[label=roadmap]{Roadmap}
  \begin{enumerate}[<+->]
    \setlength{\itemsep}{1em}
    % \alert<1,4> highlights on overlay 1 AND on the \againframe<4> return
    \item \alert<1,4>{Context}
    \item \alert<2,5>{Evidence}
    \item \alert<3,6>{Mechanism}
  \end{enumerate}
\end{frame}

\section{Context}

\begin{frame}{Takeaway or Purpose in the Title}
  Lead with the question the audience should use to interpret the evidence

  % [<+->] reveals this list one item per click
  \begin{itemize}[<+->]
    \item First context point
    \item Second context point
      \begin{itemize}[<.->]
        \item Subpoint appears with its parent, not on a separate click
      \end{itemize}
    \item Final takeaway before details
  \end{itemize}
\end{frame}

\section{Evidence}

% Roadmap return: re-show overlay 5 of the labeled frame, outside the slide count
\againframe<5>[noframenumbering]{roadmap}

\begin{frame}[label=technical-walkthrough]{Stable Technical Walkthrough}
  % overlayarea keeps the layout from jumping as content changes
  \begin{overlayarea}{\textwidth}{0.58\textheight}
    \only<1>{Start with the object the audience already understands}
    \only<2>{Introduce the new object only when needed}
    \only<3>{State the relationship the audience should notice}

    \vspace{1em}
    \[
      \visible<1->{\alert<1>{y_i}}
      \visible<2->{= \alert<2>{\alpha + \beta x_i}}
      \visible<3->{+ \alert<3>{\varepsilon_i}}
    \]

    \begin{itemize}
      \item<1-> \alert<1>{Outcome}: what we want to explain
      \item<2-> \alert<2>{Signal}: the variation we use
      \item<3-> \alert<3>{Residual}: what remains unexplained
    \end{itemize}
  \end{overlayarea}
\end{frame}

\section{Mechanism}

\againframe<6>[noframenumbering]{roadmap}

\begin{frame}{Main Slide With Backup Link}
  Keep the main path light; send expert detail to backup

  \begin{itemize}[<+->]
    \item Main claim in one line
    \item Evidence needed for the typical audience
    \item<.-> \slidelink{backup-derivation}{Derivation in backup}
  \end{itemize}
\end{frame}
